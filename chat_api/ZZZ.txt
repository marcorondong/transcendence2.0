    message: messageHandler,
    notification: notificationHandler,

export function onClientRequest(message: string, currentClient: Client): void {
    const data = parseJsonMessage(Buffer.from(message), currentClient.getSocket());
    if (!data) return;

    console.log(data);

    // Identify the handler based on the type of the data
    const handler = handlers[data.type];
    if (handler) {
        handler(data, currentClient);  // Call the appropriate handler
    }
}
